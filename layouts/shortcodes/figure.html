{{- /* Override of Hugo's internal figure shortcode. Resolves `src` against the page's bundle resources and emits an absolute URL (RelPermalink). This keeps images working when a post's English content is rendered as a fallback on the French/Arabic URLs (the image files are only published under the English bundle path). External URLs are left untouched. */ -}} {{- $src := .Get "src" -}} {{- $width := .Get "width" -}} {{- $height := .Get "height" -}} {{- with .Page.Resources.GetMatch $src -}} {{- $src = .RelPermalink -}} {{- /* Stamp intrinsic dimensions on raster images so they reserve space and don't shift layout (CLS). SVGs expose no usable .Width, so skip them. */ -}} {{- if and (not $width) (in (slice "png" "jpg" "jpeg" "gif" "webp") .MediaType.SubType) -}} {{- $width = string .Width -}}{{- $height = string .Height -}} {{- end -}} {{- end -}} {{- if .Get "link" -}} {{- end -}} {{- /* Closing img tag */ -}} {{- if .Get "link" }}{{ end -}} {{- if or (or (.Get "title") (.Get "caption")) (.Get "attr") -}} {{- end -}}
